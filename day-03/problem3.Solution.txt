/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head==nullptr){
            return head;
        }
        ListNode* curr=head;
        ListNode* next=curr->next;
        int baseVal=curr->val;
        while(next!=nullptr){
            if(next->val==baseVal){
                next=next->next;
            }else{
                curr->next=next;
                curr=curr->next;
                baseVal=curr->val;
                next=curr->next;
            }
        }
        curr->next=next;
        return head;
    }
};
